\noindent\textbf{Regents Drill}
\begin{enumerate}[itemsep=5em]
\item
Which Expression is equivalent to $2c\sqrt[3]{c}$?
\begin{enumerate}
\item$
2c^\frac{4}{3}$
\item $2c^\frac{3}{4}$
\item$
(2c)^\frac{4}{3}$
\item$
(2c)^\frac{3}{4}$
\end{enumerate}
\item
Given $m\ne0$ and $\left(17^\frac{1}{m}\right)^n = 17^2$, what is $n$ in terms of $m$?
\begin{enumerate}
\item
$2m$
\item $\frac{2}{m}$
\item $\frac{m}{2}$
\item $2^m$
\end{enumerate}
\item
Given $x\ne0$, where $m(x) = 12 x^{8a}$ and $p(x) = 3x^{2a}$, the expression $\frac{m(x)}{p(x)}$ is equivalent to
\begin{enumerate}
\item
$9x^{4a}$
\item 
$4x^6$
\item
$4x^{6a}$
\item
$4x^4$
\end{enumerate}
\item
The expression $\sqrt[3]{16x^6}$ is equivalent to
\begin{enumerate}
\item$4x^3$\item $4x^2$\item $2x^2\sqrt[3]{2}$\item $2x^3\sqrt[3]{2}$
\end{enumerate}
\item 
For $x>0$, which expression is equivalent to $\sqrt[3]{9x^2}\cdot\sqrt{9x}$?
\begin{enumerate}
\item
$9^5x^\frac{7}{2}$\item $9^6x^3$\item$9^\frac{1}{6}x^\frac{1}{3}$\item$9^\frac{5}{6}x^\frac{7}{6}$\hfill
\end{enumerate}
\end{enumerate}
