function generateNonzero(min, max)
  local output
  repeat
    output = math.random(min, max)
  until output ~= 0
  return output
end
\end{luacode*}

\begin{frame}
The standard form of a quadratic equation is $ax^2 + bx + c = 0$.
Identify the $a$, $b$ and $c$ for the following equations:
\begin{enumerate}[label=(\arabic*)]
\item $3x^2 - 2x + 4 = 0$
\vspace{\stretch{1}}
\item $-5x^2 +3 = 0$
\vspace{\stretch{1}}
\item $9x^2 +x -3 = 7$
\vspace{\stretch{1}}
\item $x^2 -2x + 3 = -x^2 -3$
\vspace{\stretch{1}}
\end{enumerate}
\pause

\color{red}
{\bf Answer:}
\begin{enumerate}[label=(\arabic*)]
\item
$a = 3,\ b =-2,\ c = 4$
\vspace{\stretch{1}}
\item
$a=-5,\ b = 0,\ c=3$
\vspace{\stretch{1}}
\item
$a=9,\ b=1,\ c=-10$
\vspace{\stretch{1}}
\item
$a=2,\ b=-2,\ c=6$
\vspace{\stretch{1}}
\end{enumerate}
\end{frame}
