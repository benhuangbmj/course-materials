\begin{definition}
A matrix $S$ is called symmetric if $S^T = S$.
\end{definition}
\pause
\begin{theorem}
Orthogonally diagonalizable matrices are symmetric.
\end{theorem}
{\bf Proof.} Suppose $A = ODO^T$, where $D$ is diagonal, and $O$ is orthogonal. Then 
\[
A^T = (ODO^T)^T = (O^T)^TD^TO^T = ODO^T = A.
\] 
\pause
\begin{theorem}
Symmetric matrices are orthogonally diagonalizable.
\end{theorem}
{\bf Proof.} The proof is highly nontrivial. Due to the lack of time, it's left to you to investigate. 
\end{frame}
\begin{frame}
However, the process of orthogonally diagonalizing a symmetric matrix is manageable:
\begin{enumerate}[label=Step \arabic*: , align=left]
\item Find the eigenvalues of the matrix.
\item Find a generic basis for each eigenspace.
\item Apply the Gram-Schmidt process to the basis of each eigenspace.
\item Put all the basis vectors side by side to form the orthogonal matrix.

\end{enumerate}
\end{frame}
