{\noindent\bf Problems.}
\begin{enumerate}
\item
Find $\displaystyle \lim_{x\to2} \frac{x^2 - 2^2}{x-2}$.
\vspace{\stretch{1}}
\item
Find $\displaystyle \lim_{x\to3} \frac{x^3 - 3^3}{x-3}$.
\vspace{\stretch{1}}
\item
Find $\displaystyle \lim_{x\to x_0} \frac{x^4 - x_0^4}{x-x_0}$.
\vspace{\stretch{1}}

\end{enumerate}
