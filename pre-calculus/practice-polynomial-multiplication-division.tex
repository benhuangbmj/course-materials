{\bf Learning Objectives:}
\begin{enumerate}
\item
I can perform polynomial multiplication.
\item 
I can perform polynomial long division.
\end{enumerate}
\noindent
{\bf Review.} Properties of Exponents:

\begin{itemize}[label=$\bullet$]
\item
$x^mx^n = x^{m+n}$
\item
$\displaystyle \frac{x^m}{x^n} = x^{m-n}$
\item
$(x^m)^n = x^{mn}$
\item 
$(xy)^m = x^my^m$
\end{itemize}
\noindent
{\bf Examples.}
\begin{enumerate}[leftmargin=*]
\item {\bf Polynomial Multiplication}:
Multiply and simplify $(3x^2+2x-2)(-x +6)$.
\begin{align*}
(3x^2+2x-2)(-x+6) &= (-3x^3 +18x^2) + (-2x^2 + 12x) + (2x -12)\\
&= -3x^3 +16x^2 + 14x -12
\end{align*}
\item {\bf Polynomial Long Division}:
 \polymul\polynomA{4x^2 - 2x + 4}{3x - 2}
 Divide \polyprint\polynomA\  by $3x - 2$.
 \begin{center}
\polylongdiv{12x^3-14x^2+16x-8}{3x-2}
\end{center}
\[
\frac{12x^3-14x^2+16x-8}{3x-2} = 4x^2 - 2x + 4
\]
\end{enumerate}
